해당 문제는 큐를 이용한 기본 연산을 알아보는 문제입니다.

문제에서 주어진 만큼 큐를 직접 만들어서 사용했습니다.

<br>

***

<br>

```java
public static class MyQueue {
    private int front, back;
    private int SIZE = 2000000;
    private int[] queue;

    MyQueue() {
        front = back = 0;
        queue = new int[SIZE];
    }

    /**
        * Queue 삽입(back)
        */
    public void push(int value) {
        queue[back] = value;
        back = (back + 1) % SIZE;
    }

    /**
        * Queue 추출(front)
        */
    public int pop() {
        int retVal = 0;
        if(front != back) {
            retVal = queue[front];
            front = (front + 1) % SIZE;
        } else {
            retVal = -1;
        }

        return retVal;
    }

    /**
        * Queue size
        */
    public int size() {
        return back >= front? back - front : SIZE - front + back;
    }

    /**
        * Queue가 비어있는지 여부
        */
    public int empty() {
        return front == back? 1 : 0;
    }

    /**
        * front 위치의 정수 출력
        */
    public int front() {
        return front == back? -1 : queue[front];
    }

    /**
        * back 위치의 정수 술력
        */
    public int back() {
        return front == back? -1 : queue[(SIZE + back - 1) % SIZE];
    }
}
```

## 문제 풀이

<br>

```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.StringTokenizer;

public class Main {
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        int N = Integer.parseInt(br.readLine());
        StringTokenizer st;
        StringBuilder sb = new StringBuilder();
        MyQueue queue = new MyQueue();

        for(int t = 0; t < N; t++) {
            st = new StringTokenizer(br.readLine(), " ");
            String cmd = st.nextToken();

            switch(cmd) {
                case "push" :
                    int value = Integer.parseInt(st.nextToken());
                    queue.push(value);
                    break;

                case "pop" :
                    sb.append(queue.pop()).append('\n');
                    break;

                case "size" :
                    sb.append(queue.size()).append('\n');
                    break;

                case "empty" :
                    sb.append(queue.empty()).append('\n');
                    break;

                case "front" :
                    sb.append(queue.front()).append('\n');
                    break;

                case "back" :
                    sb.append(queue.back()).append('\n');
                    break;
            }
        }

        System.out.println(sb);
    }

    public static class MyQueue {
        private int front, back;
        private int SIZE = 2000000;
        private int[] queue;

        MyQueue() {
            front = back = 0;
            queue = new int[SIZE];

                retVal = -1;
            }

            return retVal;
        }

        /**
         * Queue가 비어있는지 여부

        /**

        /**
         * back 위치의 정수 술력

    }
}
```
